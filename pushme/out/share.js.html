}

 * @param {String} txt 共有したいテキストデータ
 * @return {String} formattedTxt 整形済テキストデータ
 */
 function formatForSend(txt){
    var formattedTxt = "";
    formattedTxt = "今回決まったのは [" + txt + "] です！";
    return formattedTxt;
 }
